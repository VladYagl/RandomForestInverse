\newcommand{\addplotsmac}[5] {%
\addplot 
    [scatter, color=#2]
    plot [error bars/.cd, error bar style={line width=1pt}, y dir = both,
    y explicit] table[x index=#3, y index=#4, y error index=#5]{#1}; 
}
